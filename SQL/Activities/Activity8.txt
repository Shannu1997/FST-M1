Select CUSTOMER_ID, MAX(PURCHASE_AMOUNT) highPurchAmnt
    FROM orders
Group BY CUSTOMER_ID;

Select SALESMAN_ID, ORDER_DATE, MAX(PURCHASE_AMOUNT) highPurchAmnt 
    FROM orders
    where ORDER_DATE ='17-AUG-12'
Group BY SALESMAN_ID ,ORDER_DATE;

Select CUSTOMER_ID, ORDER_DATE , MAX(PURCHASE_AMOUNT) highPurchAmnt
from orders
    Group by CUSTOMER_ID, ORDER_DATE 
Having MAX(PURCHASE_AMOUNT) in (2030, 3450, 5760, 6000)
